## 実装-minHeap
- 実装時間: 15分
- 時間計算量: PriorityQueue構築O(n logk)、addメソッド1回実行 O(log k)
- 空間計算量: O(k)
- ヒープの実装を忘れてたので復習
```java
class KthLargest {
    private PriorityQueue<Integer> minHeap = new PriorityQueue<>();
    private int k = 0;

    public KthLargest(int k, int[] nums) {
        this.k = k;
        for (int num : nums) {
            add(num);
        }
    }

    public int add(int val) {
        minHeap.add(val);
        if (minHeap.size() > k) {
          minHeap.poll();
        }
        return minHeap.peek();
    }
}
```

### 復習-minHeapの実装
- [参照: Java Code To Implementation Min Heap](https://www.geeksforgeeks.org/java/heap-implementation-in-java/)
- このデータ構造の目的を理解していないと、BSTと何が違うんだっけ？ってなる
- PriorityQueue の目的は、「最小値または最大値を常にルートに持ち続けること。それ以外のノードの値の大小関係は親<=子が守られていればよく、別枝同士の大小関係は破綻していても構わない
```java
import java.util.ArrayList;

class MinHeap {
    private ArrayList<Integer> heap;

    public minHeap() {
        heap = new ArrayList<>();
    }
    
    private int parent(int i) {
        return (i - 1) / 2;
    }
    
    private int leftChild(int i) {
        return 2 * i + 1;
    }
    
    private int rightChild(int i) {
        return 2 * i + 2;
    }
    
    private void swap(int i, int j) {
        int temp = heap.get(i);
        heap.add(i, heap.get(j));
        heap.add(j, temp);
    }
    
    public void insert(int value) {
        heap.add(value);
        
        int currentIndex = heap.size() - 1;
        
        // ここがわかってなかった
        while (currentIndex > 0 && heap.get(currentIndex) < heap.get(parent(currentIndex))) {
            // 追加した末尾と、末尾の親ノードの値を比較して追加ノードのvalが小さければ交換
            swap(currentIndex, parent(currentIndex));
            // currentIndex をswap先の親Indexに更新
            currentIndex = parent(currentIndex);
        }
    }
    
    // ヒープから最小値を抽出、最後の要素をルートに持ってきて影響のある部分を再構築(狭義の heapify()?)
    public int extractMin() {
        if (heap.isEmpty()) {
            throw new RuntimeException("ヒープが空");
        }
        
        int min = heap.get(0);
        
        int lastElement = heap.remove(heap.size() - 1);
        
        if (!heap.isEmpty()) {
            // 末尾の要素をルートへ移動
            heap.set(0, lastElement);
            int currentIndex = 0;
            
            while (true) {
                int left = leftChild(currentIndex);
                int right = rightChild(currentIndex);
                
                int smallest = currentIndex;
                
                // 親ノードと子ノードを比較し、最小値のインデックスを抽出する
                // まず親と左を比較
                if (left < heap.size() && heap.get(left) < heap.get(smallest)) {
                    smallest = left;
                }
                // 次に親と左の小さい方と右を比較
                if (right < heap.size() && heap.get(right) < heap.get(smallest)) {
                    smallest = right;
                }
                // ノードの値が最も小さいインデックスと親ノードのインデックスが一致したら break
                if (smallest == currentIndex) {
                    break;    
                }
                
                swap(currentIndex, smallest);
                currentIndex = smallest;
            }
        }
        
        return min;
    }
    
    public boolean isEmpty() {
        return heap.isEmpty();
    }
}

// how to use
public class HeapExample {
    public static void main(String[] args) {
        MinHeap minHeap = new MinHeap();

        // Insert values into the min heap
        minHeap.insert(10);
        minHeap.insert(5);
        minHeap.insert(15);
        minHeap.insert(20);
        minHeap.insert(25);

        // Extract and print the minimum values from the heap
        System.out.println("Extracted Min: " + minHeap.extractMin());
        System.out.println("Extracted Min: " + minHeap.extractMin());
    }
}
```

### 別方針-1 mapで出現回数を数えてみる
- 10分くらいで書いてみたけど、余裕でTLEする
- 走査を工夫してみたが、あまり意味がない
- 時間計算量: kthLargest() --> O(N) nums.length に依存, O(20000) --> add() this.min, this.max に依存
- 空間計算量: O(N) --> nums.length に依存
```java
class KthLargest {
  private HashMap<Integer, Integer> map;
  private int k;
  private int min = -10000;
  private int max = 10000;

  public KthLargest(int k, int[] nums) {
    this.k = k;
    map = new HashMap<>();
    for (int num : nums) {
      if (map.containsKey(num)) {
        map.put(num, map.get(num) + 1);
      } else {
        map.put(num, 1);
      }

      if (num > max) {
        max = num;
      }

      if (num < min) {
        min = num;
      }
    }
  }

  public int add(int val) {
    if (map.containsKey(val)) {
      map.put(val, map.get(val) + 1);
    } else {
      map.put(val, 1);
    }

    int temp = this.k;
    for (int i = this.max; i >= this.min; i--) {
      temp -= map.getOrDefault(i, 0);
      if (temp <= 0) {
        return i;
      }
    }
    throw new IllegalStateException("これまでに追加された総数よりkが大きい");
  }
}

/**
 * Your KthLargest object will be instantiated and called as such:
 * KthLargest obj = new KthLargest(k, nums);
 * int param_1 = obj.add(val);
 */
```
## C++ バージョン
- leetcode上ではギリギリTLEしなかった
- やっていることは上記と同じ
```c++
#include <map>
using namespace std;

class KthLargest {
    map<int, int> freq; // key: 値, value: 出現回数
    int k;

public:
    KthLargest(int k, vector<int>& nums) : k(k) {
        for (int n : nums) {
            freq[n]++;
        }
    }

    int add(int val) {
        freq[val]++;

        int cnt = k;
        for (auto it = freq.rbegin(); it != freq.rend(); ++it) {
            cnt -= it->second;
            if (cnt <= 0) {
                return it->first;
            }
        }
        throw runtime_error("not enough elements");
    }
};

```
